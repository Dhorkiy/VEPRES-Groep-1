90void CreateTasks(void)
91{
92 PTASKDATA ptd = tasks; // tasks == &tasks[0]: both are addresses of first struct
93 vTaskSuspendAll(); // zet de scheduler stil en dwing FreeRTOS om eerst alle taken te starten
94
95 for (; ptd->func != NULL; ptd++) // loopje zonder teller; gewoon de pointer steeds ophogen...
96 {
97 UART_puts("\r\nCreated: ");
98 UART_puts((char *)ptd->attr.name);
99
100 // nb. kijk naar de haakjes van argument 3: het gaat om het adres van (structure attr).
101 ptd->hTask = osThreadNew(ptd->func, ptd->argument, &(ptd->attr));
102 }
103
104 xTaskResumeAll(); // start nu de scheduler: play ball
105 DisplayTaskData(); // display alle taskdata op UART
106}

162void DisplayTaskData(void)
163{
164 PTASKDATA ptd = tasks;
165 UBaseType_t freespace;
166 unsigned int used;
167 unsigned int task_nr; // tasknr for changing priority
168
169 for (task_nr=1; ptd->func != NULL; ptd++, task_nr++)
170 {
171 freespace = uxTaskGetStackHighWaterMark(ptd->hTask) * 4; // amount of free bytes
172 used = 100 - (freespace * 100) / ptd->attr.stack_size; // in percentage
173
174 UART_puts("\r\n\t["); UART_putint(task_nr); UART_puts("] ");
175 UART_puts(ptd->attr.name);
176 UART_puts("\t priority: "); UART_putint(ptd->attr.priority);
177 UART_puts("\t stacksize: "); UART_putint(ptd->attr.stack_size);
178 UART_puts("\t used: "); UART_putint(used); UART_puts("%");
179 }
180}
